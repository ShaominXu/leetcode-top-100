# [234. 回文链表](https://leetcode.cn/problems/palindrome-linked-list/)

## 思路

先用快慢指针找到中点，再反转后半段链表，随后从头和后半段头同时比较。
比较结束后可选择恢复链表（题目一般不强制）。

时间复杂度：$O(L)$。
空间复杂度：$O(1)$。

## 代码

=== "Python"
    ```python
    class Solution:
        def palindrome(self, head):
            slow = fast = head
            while fast and fast.next:
                slow = slow.next
                fast = fast.next.next
            prev = None
            cur = slow
            while cur:
                nxt = cur.next
                cur.next = prev
                prev = cur
                cur = nxt
            p1, p2 = head, prev
            while p2:
                if p1.val != p2.val:
                    return False
                p1 = p1.next
                p2 = p2.next
            return True

        def isPalindrome(self, head) -> bool:
            return self.palindrome(head)
    ```

=== "C++"
    ```cpp
    struct ListNode {
        int val;
        ListNode* next;
        ListNode(int x = 0, ListNode* n = nullptr) : val(x), next(n) {}
    };

    class Solution {
    public:
        ListNode* reverseList(ListNode* head) {
            ListNode* prev = nullptr;
            ListNode* cur = head;
            while (cur) {
                ListNode* nxt = cur->next;
                cur->next = prev;
                prev = cur;
                cur = nxt;
            }
            return prev;
        }

        bool isPalindrome(ListNode* head) {
            ListNode* slow = head;
            ListNode* fast = head;
            while (fast && fast->next) {
                slow = slow->next;
                fast = fast->next->next;
            }
            ListNode* second = reverseList(slow);
            ListNode* p1 = head;
            ListNode* p2 = second;
            while (p2) {
                if (p1->val != p2->val) return false;
                p1 = p1->next;
                p2 = p2->next;
            }
            return true;
        }
    };
    ```
