# [删除无效的括号](https://leetcode.cn/problems/remove-invalid-parentheses/)

## 思路

BFS 按删除次数逐层搜索。某层只要找到合法字符串，就停止继续向下层扩展，
因为该层删除次数最少。使用集合去重避免重复状态。

## 代码

=== "Python3"
    ```python
    from collections import deque

    class Solution:
        def removeInvalidParentheses(self, s: str) -> List[str]:
            def valid(x: str) -> bool:
                cnt = 0
                for ch in x:
                    if ch == '(':
                        cnt += 1
                    elif ch == ')':
                        cnt -= 1
                        if cnt < 0:
                            return False
                return cnt == 0

            q = deque([s])
            vis = {s}
            ans = []
            found = False
            while q:
                cur = q.popleft()
                if valid(cur):
                    ans.append(cur)
                    found = True
                if found:
                    continue
                for i, ch in enumerate(cur):
                    if ch not in '()':
                        continue
                    nxt = cur[:i] + cur[i + 1:]
                    if nxt not in vis:
                        vis.add(nxt)
                        q.append(nxt)
            return ans
    ```

=== "C++"
    ```cpp

    class Solution {
    public:
        bool valid(const string& s) {
            int cnt = 0;
            for (char ch : s) {
                if (ch == '(') ++cnt;
                else if (ch == ')') {
                    --cnt;
                    if (cnt < 0) return false;
                }
            }
            return cnt == 0;
        }

        vector<string> removeInvalidParentheses(string s) {
            queue<string> q;
            unordered_set<string> vis;
            vector<string> ans;
            q.push(s);
            vis.insert(s);
            bool found = false;
            while (!q.empty()) {
                string cur = q.front();
                q.pop();
                if (valid(cur)) {
                    ans.push_back(cur);
                    found = true;
                }
                if (found) continue;
                for (int i = 0; i < (int)cur.size(); ++i) {
                    if (cur[i] != '(' && cur[i] != ')') continue;
                    string nxt = cur.substr(0, i) + cur.substr(i + 1);
                    if (!vis.count(nxt)) {
                        vis.insert(nxt);
                        q.push(nxt);
                    }
                }
            }
            return ans;
        }
    };
    ```
