# [除法求值](https://leetcode.cn/problems/evaluate-division/)

## 思路

建图。等式 `a / b = v` 转为有向边 `a->b` 权重 `v`，`b->a` 权重 `1/v`。
对每个查询用 BFS/DFS 搜索路径，路径权重连乘即答案。
若不存在路径返回 `-1.0`。

## 代码

=== "Python3"
    ```python
    from collections import defaultdict, deque

    class Solution:
        def calcEquation(self, equations: List[List[str]], values: List[float], queries: List[List[str]]) -> List[float]:
            g = defaultdict(list)
            for (a, b), v in zip(equations, values):
                g[a].append((b, v))
                g[b].append((a, 1.0 / v))

            def bfs(s: str, t: str) -> float:
                if s not in g or t not in g:
                    return -1.0
                if s == t:
                    return 1.0
                q = deque([(s, 1.0)])
                vis = {s}
                while q:
                    x, w = q.popleft()
                    if x == t:
                        return w
                    for y, k in g[x]:
                        if y not in vis:
                            vis.add(y)
                            q.append((y, w * k))
                return -1.0

            return [bfs(a, b) for a, b in queries]
    ```

=== "C++"
    ```cpp

    class Solution {
    public:
        vector<double> calcEquation(vector<vector<string>>& equations, vector<double>& values, vector<vector<string>>& queries) {
            unordered_map<string, vector<pair<string, double>>> g;
            for (int i = 0; i < (int)equations.size(); ++i) {
                string a = equations[i][0], b = equations[i][1];
                double v = values[i];
                g[a].push_back({b, v});
                g[b].push_back({a, 1.0 / v});
            }

            auto bfs = [&](const string& s, const string& t) -> double {
                if (!g.count(s) || !g.count(t)) return -1.0;
                if (s == t) return 1.0;
                queue<pair<string, double>> q;
                unordered_set<string> vis;
                q.push({s, 1.0});
                vis.insert(s);
                while (!q.empty()) {
                    auto [x, w] = q.front();
                    q.pop();
                    if (x == t) return w;
                    for (auto& [y, k] : g[x]) {
                        if (!vis.count(y)) {
                            vis.insert(y);
                            q.push({y, w * k});
                        }
                    }
                }
                return -1.0;
            };

            vector<double> ans;
            for (auto& q : queries) ans.push_back(bfs(q[0], q[1]));
            return ans;
        }
    };
    ```

### 复杂度分析

时间复杂度：单次查询最坏 O(V+E)。
空间复杂度：O(V+E)，额外空间主要来自哈希表/队列/递归栈等辅助结构。
