# [单词搜索](https://leetcode.cn/problems/word-search/)

## 思路

回溯 + 原地标记访问。
从每个起点出发 DFS 匹配 `word[k]`，匹配后临时标记当前位置，向四邻扩展匹配下一字符，结束后恢复现场。

## 代码

=== "Python3"
    ```python

    class Solution:
        def exist(self, board: List[List[str]], word: str) -> bool:
            m, n = len(board), len(board[0])

            def dfs(i: int, j: int, k: int) -> bool:
                if k == len(word):
                    return True
                if i < 0 or i >= m or j < 0 or j >= n or board[i][j] != word[k]:
                    return False
                ch = board[i][j]
                board[i][j] = '#'
                ok = dfs(i + 1, j, k + 1) or dfs(i - 1, j, k + 1) or dfs(i, j + 1, k + 1) or dfs(i, j - 1, k + 1)
                board[i][j] = ch
                return ok

            for i in range(m):
                for j in range(n):
                    if dfs(i, j, 0):
                        return True
            return False
    ```

=== "C++"
    ```cpp

    class Solution {
    public:
        int m, n;

        bool dfs(vector<vector<char>>& board, string& word, int i, int j, int k) {
            if (k == (int)word.size()) return true;
            if (i < 0 || i >= m || j < 0 || j >= n || board[i][j] != word[k]) return false;
            char ch = board[i][j];
            board[i][j] = '#';
            bool ok = dfs(board, word, i + 1, j, k + 1) || dfs(board, word, i - 1, j, k + 1)
                   || dfs(board, word, i, j + 1, k + 1) || dfs(board, word, i, j - 1, k + 1);
            board[i][j] = ch;
            return ok;
        }

        bool exist(vector<vector<char>>& board, string word) {
            m = (int)board.size();
            n = (int)board[0].size();
            for (int i = 0; i < m; ++i) {
                for (int j = 0; j < n; ++j) {
                    if (dfs(board, word, i, j, 0)) return true;
                }
            }
            return false;
        }
    };
    ```

## 复杂度分析

时间复杂度：最坏 O(mn \cdot 4^L)，L 为单词长度。
空间复杂度：O(L)。
