53. 最大子数组和

LeetCode: 最大子数组和 · 分类：动态规划 / Kadane

给定整数数组 nums，求连续子数组之和的最大值。

核心转移：cur = max(x, cur + x)，全局答案为遍历过程中的 max(cur)。

from typing import List


class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        cur = ans = nums[0]
        for x in nums[1:]:
            cur = max(x, cur + x)
            ans = max(ans, cur)
        return ans

#include <vector>
#include <algorithm>
using namespace std;


class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int cur = nums[0], ans = nums[0];
        for (int i = 1; i < (int)nums.size(); ++i) {
            cur = max(nums[i], cur + nums[i]);
            ans = max(ans, cur);
        }
        return ans;
    }
};
